Example: nasty → nice

A 3D vector class with eight operators plus a small driver. The plain CL side uses idiomatic defstruct accessors ((vec3-x v), not the verbose (slot-value v 'x)). glu still wins because $x beats (vec3-x v), (? b x) beats (vec3-x b), and repeated defmethod heads fold into one glu form. Counts below are characters with whitespace stripped.

Plain Common Lisp — 1331 chars, 53 lines

(defparameter *config* '((scale-by 2)))

(defstruct vec3
  "3D vector."
  (x 0.0) (y 0.0) (z 0.0))

(defmethod vadd ((a vec3) (b vec3))
  (make-vec3 :x (+ (vec3-x a) (vec3-x b))
             :y (+ (vec3-y a) (vec3-y b))
             :z (+ (vec3-z a) (vec3-z b))))

(defmethod vsub ((a vec3) (b vec3))
  (make-vec3 :x (- (vec3-x a) (vec3-x b))
             :y (- (vec3-y a) (vec3-y b))
             :z (- (vec3-z a) (vec3-z b))))

(defmethod dot ((a vec3) (b vec3))
  (+ (* (vec3-x a) (vec3-x b))
     (* (vec3-y a) (vec3-y b))
     (* (vec3-z a) (vec3-z b))))

(defmethod cross ((a vec3) (b vec3))
  (make-vec3
    :x (- (* (vec3-y a) (vec3-z b)) (* (vec3-z a) (vec3-y b)))
    :y (- (* (vec3-z a) (vec3-x b)) (* (vec3-x a) (vec3-z b)))
    :z (- (* (vec3-x a) (vec3-y b)) (* (vec3-y a) (vec3-x b)))))

(defmethod scale ((v vec3) k)
  (make-vec3 :x (* (vec3-x v) k)
             :y (* (vec3-y v) k)
             :z (* (vec3-z v) k)))

(defmethod neg ((v vec3))
  (make-vec3 :x (- (vec3-x v))
             :y (- (vec3-y v))
             :z (- (vec3-z v))))

(defmethod len2 ((v vec3))
  (+ (* (vec3-x v) (vec3-x v))
     (* (vec3-y v) (vec3-y v))
     (* (vec3-z v) (vec3-z v))))

(defmethod show ((v vec3))
  (format nil "<~,2f ~,2f ~,2f>"
          (vec3-x v) (vec3-y v) (vec3-z v)))

(defmethod lerp ((a vec3) (b vec3) k)
  (vadd (scale a (- 1 k)) (scale b k)))

(defmethod vmap ((v vec3) f)
  (make-vec3 :x (funcall f (vec3-x v))
             :y (funcall f (vec3-y v))
             :z (funcall f (vec3-z v))))

(let* ((a (make-vec3 :x 1 :y 0 :z 0))
       (b (make-vec3 :x 0 :y 1 :z 0))
       (k (second (assoc 'scale-by *config*)))
       (vs (sort (list a b (cross a b) (lerp a b 0.5))
                 (lambda (x y) (< (len2 x) (len2 y))))))
  (loop for v in vs
        when (let ((n (len2 v))) (and n (> n 0)))
        do (format t "~a~%"
                   (funcall #'show
                            (vmap (scale v k)
                                  (lambda (n) (- n)))))))

glu — 813 chars, 25 lines (40% smaller)

(defparameter *config* '((scale-by 2)))
(defread @(s) `(second (assoc ',(read s t nil t) *config*)))

(glu vec3 "3D vector." ((x 0.0) (y 0.0) (z 0.0))
  (vadd (b) (new+ vec3 :x (+ $x (? b x)) :y (+ $y (? b y)) :z (+ $z (? b z))))
  (vsub (b) (new+ vec3 :x (- $x (? b x)) :y (- $y (? b y)) :z (- $z (? b z))))
  (dot  (b) (+ (* $x (? b x)) (* $y (? b y)) (* $z (? b z))))
  (cross (b)
    (new+ vec3 :x (- (* $y (? b z)) (* $z (? b y)))
              :y (- (* $z (? b x)) (* $x (? b z)))
              :z (- (* $x (? b y)) (* $y (? b x)))))
  (scale (k) (new+ vec3 :x (* $x k) :y (* $y k) :z (* $z k)))
  (neg   ()  (new+ vec3 :x (- $x) :y (- $y) :z (- $z)))
  (len2  ()  (+ (* $x $x) (* $y $y) (* $z $z)))
  (show  ()  (format nil "<~,2f ~,2f ~,2f>" $x $y $z)))

(glu+ vec3
  (lerp (b k) (vadd (scale i (- 1 k)) (scale b k)))
  (vmap (f)   (new+ vec3 :x (! f $x) :y (! f $y) :z (! f $z))))

(let+ ((a  (new+ vec3 :x 1 :y 0 :z 0))
       (b  (new+ vec3 :x 0 :y 1 :z 0))
       (vs (sort (list a b (cross a b) (lerp a b 0.5))
                 (fnn (< (len2 _) (len2 __))))))
  (for+ (format t "~a~%" (! #'show (vmap (scale v @scale-by)
                                         (fn (- _)))))
        for v in vs
        if (if+ (len2 v) (> it 0))))

Every macro, one program. defread installs the @key reader. glu bundles the struct plus eight methods. glu+ bolts lerp and vmap on later. $x reads self-slots; (? b x) reads other-slots. new+ is sugar for make-vec3. let+ sequences the bindings. fn trims the negate lambda inside vmap; fnn trims the sort comparator (fnnn works the same way with three implicit args _ __ ___). for+ walks the sorted vectors; if+ filters non-zero; ! funcalls show.
